$("#test").click(function() {
    $("#test-in").html( function() {
        test = $(this);
        $.ajax("{{ url_for('group.test', id=14) }}")
        .done(function(data) {
            test.html(data);
            
            $("#test-form").submit(function(event) {
                event.preventDefault();
                data = $(this);
                username = data.find("input[name='username'");

                
                $.ajax(
                    {
                        method: 'POST',
                        url: "{{ url_for('group.test', id=14) }}",
                        data: {'username':username.val() }
                    }
                )
                .done(function(data){
                    test.html(data);
                    
                }).fail(function(jqXHR, textStatus){
                    console.log('WAT' + textStatus);

                });
                
            });
        })

        
        
    });
});
